        - transition-timing-function(움직임 형태 제어)
           1. transition-timing-function: linear; /* 동일한 속도로 진행 */
           2. transition-timing-function: ease; /* 기본값, 서서히 증가하다가 끝에서 느려짐 */ 
           3. transition-timing-function: ease-in; /* 천천히 시작해서 끝날 때 빨라짐 */
           4. transition-timing-function: ease-out;
           5. transition-timing-function: ease-in-out;
           6. transition-timing-function: cubic-bezier(0,-0.42,.66,1.52); /* 사용자 정의 움직임 형태 */
